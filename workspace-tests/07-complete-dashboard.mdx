## ğŸ¯ Filtering & Analysis

### Large Companies (Revenue > $100B)

<DataFilter
  dataKey="companies"
  outputKey="large_companies"
  filter="item => item.revenue > 100"
  autoExecute={true}
/>

<DataList
  dataKey="large_companies"
/>

### AI Companies

<DataFilter
  dataKey="companies"
  outputKey="ai_companies"
  filter="item => item.category === 'ai'"
  autoExecute={true}
/>

<DataList
  dataKey="ai_companies"
  emptyMessage="No AI companies in dataset"
/>

### Analyze AI Companies

<DataProcessor
  dataKey="ai_companies"
  outputKey="ai_analysis"
  agentName="integration-expert"
  query="Provide a brief analysis of these AI companies"
  autoExecute={true}
/>

<MarkdownDisplay
  dataKey="ai_analysis"
  showLoading={true}
